```python
# anti_copyright/video_transformer.py

import cv2
import numpy as np
import random
from moviepy.editor import VideoFileClip, vfx

class VideoFingerprintEvasion:
    """
    Técnicas para tornar cada vídeo único digitalmente
    sem alterar qualidade visual perceptível ao humano.
    """
    
    def __init__(self, seed: int = None):
        self.seed = seed or random.randint(1, 999999)
        random.seed(self.seed)
        np.random.seed(self.seed)
    
    def apply_all(self, clip: VideoFileClip, intensity: str = "medium") -> VideoFileClip:
        """
        Aplica todas as transformações em sequência.
        intensity: low | medium | high
        """
        transformations = self._get_transformations(intensity)
        for transform in transformations:
            clip = transform(clip)
        return clip
    
    def _get_transformations(self, intensity: str) -> list:
        if intensity == "low":
            return [
                self.micro_crop,
                self.subtle_color_shift,
                self.metadata_randomize,
            ]
        elif intensity == "medium":
            return [
                self.micro_crop,
                self.subtle_color_shift,
                self.micro_speed_variation,
                self.noise_injection_imperceptible,
                self.metadata_randomize,
                self.frame_timing_jitter,
            ]
        elif intensity == "high":
            return [
                self.micro_crop,
                self.subtle_color_shift,
                self.micro_speed_variation,
                self.noise_injection_imperceptible,
                self.spatial_frequency_shift,
                self.temporal_resampling,
                self.chroma_micro_shift,
                self.metadata_randomize,
                self.frame_timing_jitter,
            ]
    
    def micro_crop(self, clip: VideoFileClip) -> VideoFileClip:
        """
        Corte micro (1-3px) em cada borda — indetectável visualmente.
        Muda hash de cada frame completamente.
        """
        w, h = clip.size
        # Variação de 1 a 3 pixels por borda
        top    = random.randint(1, 3)
        bottom = random.randint(1, 3)
        left   = random.randint(1, 3)
        right  = random.randint(1, 3)
        
        clip = clip.crop(
            x1=left,
            y1=top,
            x2=w - right,
            y2=h - bottom
        )
        # Redimensiona de volta para resolução original
        return clip.resize((w, h))
    
    def subtle_color_shift(self, clip: VideoFileClip) -> VideoFileClip:
        """
        Ajuste microscópico de RGB — imperceptível mas muda fingerprint.
        Variação máxima: ±2 em cada canal (escala 0-255).
        """
        r_shift = random.uniform(-2, 2) / 255.0
        g_shift = random.uniform(-2, 2) / 255.0
        b_shift = random.uniform(-2, 2) / 255.0
        
        def adjust_frame(frame):
            frame = frame.astype(np.float32) / 255.0
            frame[:, :, 0] = np.clip(frame[:, :, 0] + r_shift, 0, 1)
            frame[:, :, 1] = np.clip(frame[:, :, 1] + g_shift, 0, 1)
            frame[:, :, 2] = np.clip(frame[:, :, 2] + b_shift, 0, 1)
            return (frame * 255).astype(np.uint8)
        
        return clip.fl_image(adjust_frame)
    
    def micro_speed_variation(self, clip: VideoFileClip) -> VideoFileClip:
        """
        Variação de velocidade ±0.5% — imperceptível ao ouvido/olho.
        Muda completamente o fingerprint temporal.
        """
        speed_factor = random.uniform(0.995, 1.005)
        return clip.speedx(speed_factor)
    
    def noise_injection_imperceptible(self, clip: VideoFileClip) -> VideoFileClip:
        """
        Adiciona ruído gaussiano sigma=0.5 — abaixo do limiar de percepção humana.
        Cada pixel alterado em ±1 valor máximo.
        """
        def add_noise(frame):
            noise = np.random.normal(0, 0.5, frame.shape).astype(np.int16)
            frame = frame.astype(np.int16) + noise
            return np.clip(frame, 0, 255).astype(np.uint8)
        
        return clip.fl_image(add_noise)
    
    def frame_timing_jitter(self, clip: VideoFileClip) -> VideoFileClip:
        """
        Adiciona jitter de ±1ms nos timestamps dos frames.
        Muda fingerprint temporal sem afetar sincronização percebida.
        """
        jitter_ms = random.uniform(-0.001, 0.001)
        return clip.set_start(clip.start + jitter_ms)
    
    def spatial_frequency_shift(self, clip: VideoFileClip) -> VideoFileClip:
        """
        Aplica filtro de frequência espacial imperceptível usando DCT.
        Altera componentes de alta frequência (invisíveis ao olho humano).
        """
        def apply_dct_shift(frame):
            # Converte para float
            f = frame.astype(np.float32)
            # Aplica DCT 2D em cada canal
            for c in range(3):
                dct = cv2.dct(f[:, :, c])
                # Adiciona ruído apenas nas altas frequências
                h, w = dct.shape
                noise_region = np.random.uniform(-0.1, 0.1, (h//4, w//4))
                dct[3*h//4:, 3*w//4:] += noise_region
                f[:, :, c] = cv2.idct(dct)
            return np.clip(f, 0, 255).astype(np.uint8)
        
        return clip.fl_image(apply_dct_shift)
    
    def chroma_micro_shift(self, clip: VideoFileClip) -> VideoFileClip:
        """
        Deslocamento sub-pixel na crominância (YCbCr).
        Algoritmo parecido com o que JPEG usa — imperceptível.
        """
        def shift_chroma(frame):
            # Converte RGB para YCbCr
            img = cv2.cvtColor(frame, cv2.COLOR_RGB2YCrCb)
            # Aplica micro-shift apenas nos canais Cr e Cb
            shift_x = random.randint(-1, 1)
            shift_y = random.randint(-1, 1)
            M = np.float32([[1, 0, shift_x], [0, 1, shift_y]])
            h, w = img.shape[:2]
            img[:, :, 1] = cv2.warpAffine(img[:, :, 1], M, (w, h))
            img[:, :, 2] = cv2.warpAffine(img[:, :, 2], M, (w, h))
            return cv2.cvtColor(img, cv2.COLOR_YCrCb2RGB)
        
        return clip.fl_image(shift_chroma)
    
    def temporal_resampling(self, clip: VideoFileClip) -> VideoFileClip:
        """
        Re-amostragem temporal: adiciona/remove frames duplicados invisíveis.
        Muda estrutura GOP completamente.
        """
        original_fps = clip.fps
        # Converte para fps ligeiramente diferente e volta
        intermediate_fps = original_fps * random.uniform(0.998, 1.002)
        clip = clip.set_fps(intermediate_fps)
        return clip.set_fps(original_fps)
    
    def metadata_randomize(self, clip: VideoFileClip) -> VideoFileClip:
        """
        Os metadados são randomizados na etapa de output_generation.
        Aqui apenas marca o clip com seed para rastreabilidade.
        """
        clip.anti_copyright_seed = self.seed
        return clip
```

```python
# anti_copyright/audio_transformer.py

import numpy as np
import librosa
import soundfile as sf
import random
from scipy import signal

class AudioFingerprintEvasion:
    """
    Técnicas para tornar cada áudio único digitalmente.
    Todas as técnicas são imperceptíveis ao ouvido humano.
    """
    
    def apply_all(self, audio_path: str, output_path: str, intensity: str = "medium"):
        y, sr = librosa.load(audio_path, sr=None, mono=False)
        
        if intensity in ["medium", "high"]:
            y = self.pitch_micro_shift(y, sr)
            y = self.time_stretch_imperceptible(y)
            y = self.noise_floor_injection(y)
            y = self.eq_micro_variation(y, sr)
        
        if intensity == "high":
            y = self.stereo_micro_panning(y)
            y = self.harmonic_injection(y, sr)
            y = self.reverb_imperceptible(y, sr)
        
        sf.write(output_path, y.T if y.ndim > 1 else y, sr)
        return output_path
    
    def pitch_micro_shift(self, y, sr):
        """
        Pitch shift de ±0.05 semitones — imperceptível ao ouvido humano.
        Limiar de percepção humana: ~5 cents = 0.05 semitones.
        """
        n_steps = random.uniform(-0.05, 0.05)
        if y.ndim > 1:
            return np.array([librosa.effects.pitch_shift(ch, sr=sr, n_steps=n_steps) for ch in y])
        return librosa.effects.pitch_shift(y, sr=sr, n_steps=n_steps)
    
    def time_stretch_imperceptible(self, y):
        """
        Time stretch ±0.3% — imperceptível mas muda fingerprint temporal.
        """
        rate = random.uniform(0.997, 1.003)
        if y.ndim > 1:
            return np.array([librosa.effects.time_stretch(ch, rate=rate) for ch in y])
        return librosa.effects.time_stretch(y, rate=rate)
    
    def noise_floor_injection(self, y):
        """
        Injeta ruído branco -80dB — abaixo do piso de ruído típico.
        -80dB = amplitude 0.0001 em sinal normalizado.
        """
        noise_amplitude = 0.0001 * random.uniform(0.5, 1.5)
        noise = np.random.normal(0, noise_amplitude, y.shape)
        return np.clip(y + noise, -1.0, 1.0)
    
    def eq_micro_variation(self, y, sr):
        """
        EQ de ±0.5dB em 3 bandas aleatórias — imperceptível.
        """
        frequencies = [200, 1000, 4000, 8000, 16000]
        for freq in random.sample(frequencies, 3):
            gain_db = random.uniform(-0.5, 0.5)
            gain_linear = 10 ** (gain_db / 20)
            b, a = signal.butter(2, freq / (sr / 2), btype='high')
            filtered = signal.filtfilt(b, a, y)
            y = y + (gain_linear - 1) * filtered
        return np.clip(y, -1.0, 1.0)
    
    def stereo_micro_panning(self, y):
        """
        Micro-ajuste de panorama estéreo ±1% — imperceptível.
        Só aplicado se o áudio for estéreo.
        """
        if y.ndim < 2 or y.shape[0] < 2:
            return y
        pan = random.uniform(-0.01, 0.01)
        y[0] = y[0] * (1 + pan)
        y[1] = y[1] * (1 - pan)
        return np.clip(y, -1.0, 1.0)
    
    def harmonic_injection(self, y, sr):
        """
        Injeta harmônicos de amplitude -60dB — abaixo do limiar de audição.
        """
        harmonic_amplitude = 0.001 * random.uniform(0.5, 1.5)
        harmonic_freq = random.choice([3000, 5000, 7000, 9000])
        t = np.linspace(0, len(y) / sr, len(y) if y.ndim == 1 else y.shape[1])
        harmonic = harmonic_amplitude * np.sin(2 * np.pi * harmonic_freq * t)
        if y.ndim > 1:
            harmonic = np.array([harmonic, harmonic])
        return np.clip(y + harmonic, -1.0, 1.0)
    
    def reverb_imperceptible(self, y, sr):
        """
        Reverb de 5ms com decay -40dB — cria eco imperceptível
        mas muda fingerprint de correlação temporal.
        """
        delay_samples = int(0.005 * sr)  # 5ms
        decay = 0.01
        impulse = np.zeros(delay_samples + 1)
        impulse[0] = 1.0
        impulse[-1] = decay
        if y.ndim > 1:
            return np.array([np.convolve(ch, impulse, mode='same') for ch in y])
        return np.convolve(y, impulse, mode='same')
```

```python
import random
import string
import hashlib
from datetime import datetime, timedelta

class MetadataRandomizer:
    """
    Randomiza metadados EXIF/ID3/MP4 para evitar detecção por metadados.
    Cada export tem metadados únicos.
    """
    
    def get_ffmpeg_metadata_args(self, project_id: str) -> list:
        """
        Retorna argumentos FFmpeg para injetar metadados randomizados.
        """
        seed = int(hashlib.md5(project_id.encode()).hexdigest()[:8], 16)
        rng = random.Random(seed)
        
        # Timestamp falso dentro de range plausível (últimos 2 anos)
        fake_date = datetime.now() - timedelta(
            days=rng.randint(0, 730),
            hours=rng.randint(0, 23),
            minutes=rng.randint(0, 59)
        )
        
        # Tool string aleatória (oculta que foi gerado pelo ClipFusion)
        tools = [
            "Adobe Premiere Pro 2024",
            "DaVinci Resolve 18",
            "Final Cut Pro 10.7",
            "CapCut Desktop 3.0",
            "HandBrake 1.7.0",
        ]
        
        # Encoder string randomizada
        encoders = [
            "libx264 (High@L4.1)",
            "libx265 (Main@L4.0)",
            "H.264/AVC Codec",
        ]
        
        return [
            "-metadata", f"creation_time={fake_date.isoformat()}",
            "-metadata", f"encoder={rng.choice(encoders)}",
            "-metadata", f"comment={self._random_comment(rng)}",
            "-metadata", "copyright=",
            "-metadata", "description=",
            "-metadata", "title=",
            "-metadata", "artist=",
            "-metadata", "album=",
        ]
    
    def _random_comment(self, rng) -> str:
        """Gera comentário técnico plausível mas aleatório."""

            f"Processed with color grade",
            f"Edit v{rng.randint(1,5)}.{rng.randint(0,9)}",
            "",
        ]
        return rng.choice(comments)
    
    def strip_all_metadata_ffmpeg_args(self) -> list:
        """
        Remove TODOS os metadados identificáveis.
        Usar antes de injetar os novos.
        """
        return ["-map_metadata", "-1"]
```

```python
# anti_copyright/upload_scheduler.py

import random
from datetime import datetime, timedelta
from typing import List

class AntiPatternUploadScheduler:
    """
    Agenda uploads evitando padrões detectáveis pelas plataformas.
    
    Plataformas detectam:
    - Uploads em intervalos regulares (ex: exatamente a cada 24h)
    - Uploads sempre no mesmo horário
    - Múltiplos uploads em sequência rápida
    - Fingerprint de rede (mesmo IP, mesmo user-agent)
    """
    
    def generate_upload_schedule(
        self, 
        video_count: int,
        platform: str,
        strategy: str = "natural"
    ) -> List[datetime]:
        """
        Gera agenda com variação natural humana.
        strategy: natural | aggressive | conservative
        """
        schedule = []
        base_time = datetime.now()
        
        if strategy == "natural":
            # Simula comportamento de criador real:
            # - Posts em horários de pico (18h-21h)
            # - 1-2 posts por dia
            # - Variação de ±30-90min por post
            # - Fins de semana diferentes dos dias de semana
            
            intervals = self._natural_intervals(video_count, platform)
            current = base_time
            
            for interval_hours in intervals:
                # Adiciona jitter aleatório ±20%
                jitter = interval_hours * random.uniform(-0.2, 0.2)
                current = current + timedelta(hours=interval_hours + jitter)
                
                # Ajusta para horário de pico da plataforma
                current = self._adjust_to_peak_hour(current, platform)
                schedule.append(current)
        
        return schedule
    
    def _natural_intervals(self, count: int, platform: str) -> List[float]:
        """
        Intervalos naturais por plataforma.
        TikTok: pode postar 3-5x/dia
        Instagram: 1-2x/dia
        YouTube: 1-3x/semana
        """
        if platform == "tiktok":
            # 4-8h entre posts, 3-5 posts/dia
            return [random.uniform(4, 8) for _ in range(count)]
        elif platform == "instagram":
            # 18-30h entre posts
            return [random.uniform(18, 30) for _ in range(count)]
        elif platform == "youtube":
            # 48-96h entre posts
            return [random.uniform(48, 96) for _ in range(count)]
        else:
            return [random.uniform(12, 48) for _ in range(count)]
    
    def _adjust_to_peak_hour(self, dt: datetime, platform: str) -> datetime:
        """
        Ajusta horário para janela de pico, mantendo variação humana.
        """
        peak_hours = {
            "tiktok":    [(7, 9), (12, 14), (19, 22)],
            "instagram": [(11, 13), (19, 21)],
            "youtube":   [(14, 16), (19, 21)],
        }
        windows = peak_hours.get(platform, [(18, 21)])
        window = random.choice(windows)
        
        target_hour = random.uniform(window[0], window[1])
        target_minute = random.randint(0, 59)
        
        return dt.replace(
            hour=int(target_hour),
            minute=target_minute,
            second=random.randint(0, 59)
        )
```

```python
            # Limpa temporários
            import shutil
            shutil.rmtree(temp_dir, ignore_errors=True)
        
        return result
    
    def _build_ffmpeg_command(
        self, video_path, audio_path, output_path, config, platform
    ) -> list:
        """Monta comando FFmpeg otimizado por plataforma."""
        
        w, h = config["width"], config["height"]
        
        # Filtro de escala e pad para aspect ratio
        scale_filter = (
            f"scale={w}:{h}:force_original_aspect_ratio=decrease,"
            f"pad={w}:{h}:(ow-iw)/2:(oh-ih)/2:black"
        )
        
        cmd = ["ffmpeg", "-y"]
        
        # Input vídeo
        cmd += ["-i", video_path]
        # Input áudio
        cmd += ["-i", audio_path]
        
        # VA-API hardware acceleration (Intel)
        if self.use_vaapi:
            cmd = ["ffmpeg", "-y",
                   "-hwaccel", "vaapi",
                   "-hwaccel_output_format", "vaapi",
                   "-vaapi_device", "/dev/dri/renderD128",
                   "-i", video_path,
                   "-i", audio_path]
            scale_filter = (
                f"hwupload,scale_vaapi=w={w}:h={h}"
            )
        
        # Filtros de vídeo
        cmd += ["-filter_complex", f"[0:v]{scale_filter}[v]"]
        cmd += ["-map", "[v]", "-map", "1:a"]
        
        # Codec vídeo
        if self.use_vaapi:
            cmd += ["-c:v", "h264_vaapi"]
            cmd += ["-qp", str(config.get("crf", 22))]
        else:
            cmd += ["-c:v", "libx264"]
            cmd += ["-preset", config["preset"]]
            cmd += ["-crf", str(config["crf"])]
            cmd += ["-profile:v", config["profile"]]
            cmd += ["-level", config["level"]]
        
        # Codec áudio
        cmd += ["-c:a", "aac"]
        cmd += ["-b:a", config["audio_bitrate"]]
        cmd += ["-ar", "44100"]
        
        # FPS e formato de pixel
        cmd += ["-r", str(config["fps"])]
        cmd += ["-pix_fmt", config["pixel_format"]]
        
        # Bitrate máximo
        cmd += ["-maxrate", config["video_bitrate"]]
        cmd += ["-bufsize", str(int(config["video_bitrate"][:-1]) * 2) + "k"]
        
        # Duração máxima
        if config.get("max_duration"):
            cmd += ["-t", str(config["max_duration"])]
        
        # Metadados Anti-Copyright (remove originais e injeta novos)
        cmd += self.metadata_randomizer.strip_all_metadata_ffmpeg_args()
        cmd += self.metadata_randomizer.get_ffmpeg_metadata_args(self.project_id)
        
        # Fast start para streaming
        cmd += ["-movflags", "+faststart"]
        
        cmd += [output_path]
        return cmd
    
    def _verify_output(self, output_path: str, config: dict) -> dict:
        """Verifica qualidade do output com ffprobe."""
        cmd = [
            "ffprobe", "-v", "quiet",
            "-print_format", "json",
            "-show_format", "-show_streams",
            output_path
        ]
        result = subprocess.run(cmd, capture_output=True, text=True)
        probe = json.loads(result.stdout)
        
        metrics = {
            "file_size_mb": os.path.getsize(output_path) / (1024*1024),
            "duration": float(probe["format"].get("duration", 0)),
            "bitrate_kbps": int(probe["format"].get("bit_rate", 0)) // 1000,
        }
        
        for stream in probe.get("streams", []):
            if stream["codec_type"] == "video":
                metrics["video_codec"] = stream["codec_name"]
                metrics["resolution"] = f"{stream['width']}x{stream['height']}"
                metrics["fps"] = eval(stream.get("r_frame_rate", "30/1"))
            elif stream["codec_type"] == "audio":
                metrics["audio_codec"] = stream["codec_name"]
                metrics["sample_rate"] = stream.get("sample_rate")
        
        return metrics
```

```python
import torch

from keybert import KeyBERT
import numpy as np
from functools import lru_cache
from typing import Dict, List, Optional
import threading

class NarrativeEngineV2:
    """
    Engine com lazy loading e cache de modelos.
    Evita recarregar 4GB de modelos a cada request.
    Thread-safe.
    """
    
    _instance = None
    _lock = threading.Lock()
    _models_loaded = False
    
    # Singleton thread-safe
    def __new__(cls):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super().__new__(cls)
        return cls._instance
    
    def _load_models(self):
        """Carrega modelos uma única vez."""
        if self._models_loaded:
            return
        
        with self._lock:
            if self._models_loaded:
                return
            
            device = "cuda" if torch.cuda.is_available() else "cpu"
            
            # Sentimento
            self.sentiment_pipeline = pipeline(
                "sentiment-analysis",
                model="distilbert-base-uncased-finetuned-sst-2-english",
                device=device
            )
            
            # Emoção
            self.emotion_pipeline = pipeline(
                "text-classification",
                model="j-hartmann/emotion-english-distilroberta-base",
                top_k=None,
                device=device
            )
            
            # Keywords
            self.kw_model = KeyBERT("all-MiniLM-L6-v2")
            
            self._models_loaded = True
    
    def analyze(self, text: str) -> Dict:
        """Analisa texto e mapeia para arquétipos."""
        self._load_models()
        
        # Divide em segmentos de ~100 palavras
        segments = self._split_into_segments(text)
        
        results = []
        for i, segment in enumerate(segments):
            archetype = self._map_to_archetype(segment, i)
            results.append(archetype)
        
        return {
            "project_id": self._generate_project_id(text),
            "segments": results,
            "total_duration_estimate": len(segments) * 5  # 5s por segmento
        }
    
    def _map_to_archetype(self, text: str, position: int) -> Dict:
        """Mapeia segmento para arquétipo com voice config."""
        
        sentiment = self.sentiment_pipeline(text[:512])[0]
        emotions = {e["label"]: e["score"] 
                   for e in self.emotion_pipeline(text[:512])[0]}
        keywords = self.kw_model.extract_keywords(text, top_n=3)
        
        archetype_id = self._select_archetype(sentiment, emotions, position)
        voice_config = self._get_voice_config(archetype_id, emotions)
        visual_metaphor = self._get_visual_metaphor(archetype_id, keywords)
        
        return {
            "segment_id": position + 1,
            "archetype": archetype_id,
            "text": text,
            "emotional_profile": {
                "valence": (sentiment["score"] if sentiment["label"] == "POSITIVE" 
                           else -sentiment["score"]),
                "arousal": emotions.get("fear", 0) + emotions.get("anger", 0) * 0.5
            },
            "voice_config": voice_config,
            "visual_metaphor": visual_metaphor,
            "soundscape": self._get_soundscape(archetype_id)
        }
    
    def _select_archetype(self, sentiment, emotions, position: int) -> str:
        """
        Lógica de seleção de arquétipo baseada em posição + emoção.
        """
        archetypes = {
            0: "01_despertar",
            1: "02_tensao",
            2: "03_confronto",
            3: "04_virada",
            4: "05_revelacao",
            5: "06_justo_engolido",
            6: "07_transformacao",
            7: "08_resolucao",
            8: "09_impacto",
            9: "10_encerramento"
        }
        
        # Base: posição no segmento
        base = min(position, 9)
        
        # Ajuste por emoção dominante
        dominant_emotion = max(emotions, key=emotions.get)
        
        if dominant_emotion == "anger" and emotions["anger"] > 0.6:
            return "03_confronto"
        elif dominant_emotion == "joy" and emotions["joy"] > 0.7:
            return "04_virada"
        elif dominant_emotion == "fear" and emotions["fear"] > 0.5:
            return "02_tensao"
        elif dominant_emotion == "surprise" and emotions["surprise"] > 0.6:
            return "05_revelacao"
        
        return archetypes[base]
    
    def _get_voice_config(self, archetype_id: str, emotions: Dict) -> Dict:
        VOICE_MAP = {
            "01_despertar":   {"voice": "antony", "speed": 0.9, "tone": "thoughtful"},

            "05_revelacao":   {"voice": "aria",   "speed": 0.95,"tone": "mysterious"},

        }
        return VOICE_MAP.get(archetype_id, {"voice": "antony", "speed": 1.0, "tone": "neutral"})
    
    def _get_visual_metaphor(self, archetype_id: str, keywords: List) -> str:
```

```python
# file_watcher/watcher.py

from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler, FileCreatedEvent
import subprocess
import json
import redis
import logging
from pathlib import Path
from dataclasses import dataclass
from typing import Dict, Optional
import time

@dataclass
class BatchStatus:
    project_id: str
    expected_video_count: int
    expected_audio_count: int
    received_videos: list
    received_audios: list
    validated: bool = False

class ClipFusionFileHandler(FileSystemEventHandler):
    
    def __init__(self, redis_client: redis.Redis, config: dict):
        self.redis = redis_client
        self.config = config
        self.logger = logging.getLogger(__name__)
        self._batch_cache: Dict[str, BatchStatus] = {}
    
    def on_created(self, event):
        if event.is_directory:
            return
        self._process_new_file(Path(event.src_path))
    
    def on_moved(self, event):
        """Também trata arquivos movidos para a pasta."""
        if not event.is_directory:
            self._process_new_file(Path(event.dest_path))
    
    def _process_new_file(self, file_path: Path):
        """Processa arquivo novo com validação completa."""
        
        # Aguarda arquivo terminar de ser escrito (importante!)
        self._wait_for_file_ready(file_path)
        
        # Identifica projeto a partir do path
        project_id = self._extract_project_id(file_path)
        if not project_id:
            return
        
        # Valida arquivo
        validation_result = self._validate_file(file_path)
        if not validation_result["valid"]:
            self._handle_validation_error(file_path, validation_result)
            return
        
        # Registra no batch
        self._register_file_in_batch(project_id, file_path)
        
        # Verifica se batch está completo
        if self._is_batch_complete(project_id):
            self._trigger_processing(project_id)
    
    def _wait_for_file_ready(self, path: Path, timeout: int = 60):
        """
        Aguarda arquivo parar de crescer (upload/cópia em andamento).
        Verifica tamanho a cada 1s até estabilizar.
        """
        last_size = -1
        stable_count = 0
        start = time.time()
        
        while time.time() - start < timeout:
            try:
                current_size = path.stat().st_size
                if current_size == last_size:
                    stable_count += 1
                    if stable_count >= 3:  # Estável por 3s
                        return
                else:
                    stable_count = 0
                last_size = current_size
            except FileNotFoundError:
                return
            time.sleep(1)
    
    def _validate_file(self, file_path: Path) -> dict:
        """Valida arquivo com ffprobe."""
        
        suffix = file_path.suffix.lower()
        
        is_video = suffix in [".mp4", ".mov", ".avi", ".mkv"]
        is_audio = suffix in [".wav", ".mp3", ".flac", ".aac"]
        
        if not (is_video or is_audio):
            return {"valid": False, "error": "formato_nao_suportado"}
        
        # ffprobe
        cmd = [
            "ffprobe", "-v", "quiet",
            "-print_format", "json",
            "-show_format", "-show_streams",
            str(file_path)
        ]
        
        result = subprocess.run(cmd, capture_output=True, text=True, timeout=30)
        if result.returncode != 0:
            return {"valid": False, "error": "arquivo_corrompido"}
        
        try:
            probe = json.loads(result.stdout)
        except json.JSONDecodeError:
            return {"valid": False, "error": "probe_falhou"}
        
        duration = float(probe["format"].get("duration", 0))
        
        # Valida duração (4.5s a 5.5s para vídeos de 5s)
        if is_video:
            if not (4.5 <= duration <= 5.5):
                return {
                    "valid": False, 
                    "error": f"duracao_invalida: {duration:.2f}s (esperado 4.5-5.5s)"
                }
            
            # Verifica resolução mínima
            for stream in probe.get("streams", []):
                if stream["codec_type"] == "video":
                    if stream["width"] < 1280 or stream["height"] < 720:
                        return {
                            "valid": False,
                            "error": f"resolucao_baixa: {stream['width']}x{stream['height']}"
                        }
        
        return {"valid": True, "duration": duration, "probe": probe}
    
    def _trigger_processing(self, project_id: str):
        """Dispara task no Celery via Redis."""
        task_data = {
            "project_id": project_id,
            "batch": self._batch_cache[project_id].__dict__,
            "triggered_at": time.time()
        }
        
        self.redis.lpush("render_queue", json.dumps(task_data))
        self.logger.info(f"Batch completo, disparando render: {project_id}")
        
        # Atualiza status no Redis
        self.redis.hset(f"project:{project_id}", "status", "rendering")
    
    def _handle_validation_error(self, file_path: Path, result: dict):
        error_data = {
            "file": str(file_path),
            "error": result["error"],
            "timestamp": time.time()
        }
        self.redis.lpush("validation_errors", json.dumps(error_data))
        self.logger.warning(f"Arquivo inválido: {file_path} - {result['error']}")
    
    def _extract_project_id(self, file_path: Path) -> Optional[str]:
        parts = file_path.parts
        for i, part in enumerate(parts):
            if part.startswith("clipfusion_project_"):
                return part
        return None
    
    def _register_file_in_batch(self, project_id: str, file_path: Path):
        if project_id not in self._batch_cache:
            config = self._load_project_config(project_id)
            self._batch_cache[project_id] = BatchStatus(
                project_id=project_id,
                expected_video_count=config.get("expected_files_count", 10),
                expected_audio_count=config.get("expected_files_count", 10),
                received_videos=[],
                received_audios=[]
            )
        
        batch = self._batch_cache[project_id]
        suffix = file_path.suffix.lower()
        
        if suffix in [".mp4", ".mov", ".avi"]:
            if str(file_path) not in batch.received_videos:
                batch.received_videos.append(str(file_path))
        elif suffix in [".wav", ".mp3", ".flac"]:
            if str(file_path) not in batch.received_audios:
                batch.received_audios.append(str(file_path))
    
    def _is_batch_complete(self, project_id: str) -> bool:
        batch = self._batch_cache.get(project_id)
        if not batch:
            return False
        return (len(batch.received_videos) >= batch.expected_video_count and
                len(batch.received_audios) >= batch.expected_audio_count)
    
    def _load_project_config(self, project_id: str) -> dict:
        config_path = Path(f"/projects/{project_id}/config.yaml")
        if config_path.exists():
            import yaml
            with open(config_path) as f:
                return yaml.safe_load(f)
        return {}


def start_watcher(watch_root: str, redis_url: str):
    r = redis.from_url(redis_url)
    handler = ClipFusionFileHandler(r, {})
    observer = Observer()
    observer.schedule(handler, watch_root, recursive=True)
    observer.start()
    return observer
```
