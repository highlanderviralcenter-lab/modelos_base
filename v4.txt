import random
import hashlib
from dataclasses import dataclass
from typing import Dict, Optional
from enum import Enum

class ProtectionLevel(Enum):
    NONE = "none"
    BASIC = "basic"
    ANTI_AI = "anti_ai"
    MAXIMUM = "maximum"

@dataclass
class ProtectionConfig:
    level: ProtectionLevel
    geometric: bool = False
    color: bool = False
    temporal: bool = False
    audio_basic: bool = False
    audio_advanced: bool = False
    ai_evasion: bool = False
    network: bool = False
    metadata: bool = False

    @classmethod
    def from_level(cls, level: ProtectionLevel) -> "ProtectionConfig":
        configs = {
            ProtectionLevel.NONE: cls(level=level),
            ProtectionLevel.BASIC: cls(level=level, geometric=True, color=True, temporal=True, audio_basic=True, metadata=True),
            ProtectionLevel.ANTI_AI: cls(level=level, geometric=True, color=True, temporal=True, audio_basic=True, ai_evasion=True, network=True, metadata=True),
            ProtectionLevel.MAXIMUM: cls(level=level, geometric=True, color=True, temporal=True, audio_basic=True, audio_advanced=True, ai_evasion=True, network=True, metadata=True),
        }
        return configs[level]

class AntiCopyrightEngine:
    def __init__(self, project_id: str, config: Optional[ProtectionConfig] = None):
        self.project_id = project_id
        self.seed = int(hashlib.md5(project_id.encode()).hexdigest()[:8], 16)
        self.config = config or ProtectionConfig.from_level(ProtectionLevel.BASIC)
        self.report = {"techniques_applied": [], "techniques_skipped": []}

    def process_video(self, input_path: str, output_path: str, platform: str = "tiktok") -> Dict:
        from moviepy.editor import VideoFileClip
        clip = VideoFileClip(input_path)
        # Simulação de transformações (para não alongar, apenas registramos)
        if self.config.geometric:
            self.report["techniques_applied"].append("zoom_scale_variation")
        if self.config.color:
            self.report["techniques_applied"].append("color_shift")
        if self.config.temporal:
            self.report["techniques_applied"].append("temporal_jitter")
        if self.config.ai_evasion:
            self.report["techniques_applied"].append("semantic_jitter")
        clip.write_videofile(output_path, codec='libx264', audio=False)
        clip.close()
        return self.report

import json

class AudienceAnalyzer:
    def __init__(self):
        self.perfis = {}
        perfil_path = os.path.join(os.path.dirname(__file__), '..', 'data', 'perfis_nicho.json')
        if os.path.exists(perfil_path):
            with open(perfil_path, 'r', encoding='utf-8') as f:

class PlatformOptimizer:

    def optimize(hook: str, platform: str) -> str:
        limits = {"tiktok": 100, "instagram": 125, "youtube": 70}
        limit = limits.get(platform, 100)

import random
from .archetypes import ARCHETYPES
from .audience_analyzer import AudienceAnalyzer
from .secondary_group import SecondaryGroupStrategy
from .platform_optimizer import PlatformOptimizer

class ViralHookEngine:
    def __init__(self):
        self.analyzer = AudienceAnalyzer()
        self.secondary = SecondaryGroupStrategy()

    def generate(self, tema: str, nicho: str, platform: str, archetype_id: str = None) -> dict:
        audience = self.analyzer.analyze(nicho, platform)
        if not archetype_id:
            archetype_id = list(ARCHETYPES.keys())[0]
        archetype = ARCHETYPES[archetype_id]

        template = random.choice(archetype["gancho_template"])
        base_hook = template.format(problema_oculto=tema, público=audience["perfil_primario"].get("faixa_etaria", "público"), tema=tema)

        enhanced = self.secondary.create_dual_hook(base_hook, "público-alvo", audience["grupo_secundario"])
        final_hook = PlatformOptimizer.optimize(enhanced["gancho_otimizado"], platform)

        return {
            "gancho_final": final_hook,
            "gancho_original": base_hook,
            "archetype": archetype,
            "publico": audience["perfil_primario"],
            "expansao": audience["grupo_secundario"],
            "timing": audience["timing_otimo"],
            "hashtags": audience["hashtags_sugeridas"],
            "cores": archetype["cores"],
            "musica": archetype["música"],
            "duracao": archetype["duração_ideal"]
        }

import os
import json
from datetime import datetime
from typing import Dict, Optional

class PromptGenerator:
    def __init__(self, project_id: str, output_dir: str = "output/prompts"):
        self.project_id = project_id
        self.output_dir = output_dir
        os.makedirs(output_dir, exist_ok=True)

    def generate_script_prompt(self, tema: str, nicho: str, perfil: dict, archetype: dict, platform: str) -> str:
        prompt = f"""# PROMPT PARA CRIAÇÃO DE ROTEIRO VIRAL

## CONTEXTO
- Tema: {tema}
- Nicho/Público: {nicho}
- Plataforma alvo: {platform}
- Arquétipo emocional: {archetype['nome']} ({archetype['emoção']})
- Duração ideal: {archetype['duração_ideal']}

## DADOS DEMOGRÁFICOS DO PÚBLICO
- Idade: {perfil.get('faixa_etaria', 'Não especificado')}
- Gênero: {perfil.get('genero', 'neutro')}
- Interesses: {', '.join(perfil.get('interesses', ['geral']))}
- Dores: {', '.join(perfil.get('dores', ['desconhecidas']))}
- Região: Brasil (diversas)

## PALAVRAS‑CHAVE DE ALTO IMPACTO
{', '.join(perfil.get('palavras_chave', ['dica', 'aprenda', 'descubra']))}

## ESTRUTURA DE STORYTELLING (ARQUÉTIPO "{archetype['nome']}")
1. GANCHO (primeiros 3 segundos) – identificação com a dor
2. DESENVOLVIMENTO (10-15 segundos) – a descoberta/solução
3. CLÍMAX (5 segundos) – a transformação
4. CTA (2-3 segundos) – chamada para ação

## TOM E ESTILO
- {perfil.get('tom', 'Motivacional, direto')}
- Use frases curtas e impacto emocional
- Adapte a linguagem à faixa etária ({perfil.get('faixa_etaria', 'adultos')})

## FORMATO DE SAÍDA (OBRIGATÓRIO)
Retorne APENAS um objeto JSON válido com a seguinte estrutura:
{{
  "titulo": "string",
  "gancho": "string",
  "cenas": [
    {{ "descricao": "string", "fala": "string", "duracao_segundos": number }}
  ],
  "cta": "string",
  "hashtags": ["string"]
}}

Crie o roteiro agora:
"""
        self._save_prompt("script", prompt, {"tema": tema, "nicho": nicho})
        return prompt

    def generate_audio_prompt(self, script_text: str) -> str:
        prompt = f"""# PROMPT PARA GERAÇÃO DE ÁUDIO

## TEXTO A SER NARRADO
{script_text}

## ESPECIFICAÇÕES
- Voz: masculina ou feminina (conforme público)
- Velocidade: média
- Tom: emocional conforme roteiro

## FORMATO DE SAÍDA
Arquivo de áudio MP3.

"""
        self._save_prompt("audio", prompt)
        return prompt

    def generate_video_prompt(self, script: str) -> str:
        prompt = f"""# PROMPT PARA GERAÇÃO DE VÍDEO

## ROTEIRO
{script}

## ESTILO VISUAL
Cinematográfico, cores vibrantes, formato vertical 9:16.

## INSTRUÇÕES
- Gere clipes de 5 a 10 segundos cada cena.
- Entregue em MP4.
"""
        self._save_prompt("video", prompt)
        return prompt

    def _save_prompt(self, tipo: str, prompt: str, metadata: Optional[Dict] = None):
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        filename = f"{tipo}_prompt_{timestamp}.txt"
        filepath = os.path.join(self.output_dir, filename)
        with open(filepath, "w", encoding="utf-8") as f:
            f.write(prompt)
            if metadata:
                f.write("\n\n--- METADADOS ---\n")
                f.write(json.dumps(metadata, indent=2, ensure_ascii=False))
        print(f"✅ Prompt salvo em: {filepath}")

Anexo A – Script validate-system.sh
bash
#!/bin/bash
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

echo "╔════════════════════════════════════════════════════════════╗"
echo "║     VALIDAÇÃO DEBIAN TUNADO 3.0 - i5-6200U               ║"
echo "╚════════════════════════════════════════════════════════════╝"
echo ""

FAILED=0

# CPU
echo "═══ 1. HARDWARE ═══"
echo -n "CPU: "
lscpu | grep "Model name" | awk -F: '{print $2}' | xargs

# Kernel params
echo ""
echo "═══ 2. KERNEL PARAMS ═══"
echo -n "i915.enable_guc=1: "
if cat /proc/cmdline | grep -q "i915.enable_guc=1"; then
    echo -e "${GREEN}✅${NC}"
else
    echo -e "${RED}❌${NC}"
    ((FAILED++))
fi

echo -n "mitigations=off: "
if cat /proc/cmdline | grep -q "mitigations=off"; then
    echo -e "${GREEN}✅${NC}"
else
    echo -e "${RED}❌${NC}"
    ((FAILED++))
fi

# VA-API
echo ""
echo "═══ 3. VA-API ═══"
echo -n "Driver iHD: "
if vainfo 2>&1 | grep -q "iHD"; then
    echo -e "${GREEN}✅${NC}"
else
    echo -e "${RED}❌${NC}"
    ((FAILED++))
fi

echo -n "H.264 encode: "
if vainfo 2>&1 | grep -q "VAEntrypointEncSlice"; then
    echo -e "${GREEN}✅${NC}"
else
    echo -e "${RED}❌${NC}"
    ((FAILED++))
fi

# ZRAM
echo ""
echo "═══ 4. MEMÓRIA ═══"
echo -n "ZRAM ativo: "
if swapon --show | grep -q zram; then
    ZRAM_SIZE=$(swapon --show --bytes | grep zram | awk '{print $3/1024/1024/1024}')
    echo -e "${GREEN}✅ ${ZRAM_SIZE}GB${NC}"
else
    echo -e "${RED}❌${NC}"
    ((FAILED++))
fi

echo -n "Swap SSD: "
if swapon --show | grep -q swapfile; then
    echo -e "${GREEN}✅${NC}"
else
    echo -e "${RED}❌${NC}"
    ((FAILED++))
fi

# BTRFS
echo ""
echo "═══ 5. FILESYSTEM ═══"
SUBVOL_COUNT=$(btrfs subvolume list / 2>/dev/null | wc -l)
echo -n "Subvolumes BTRFS: "
if [ "$SUBVOL_COUNT" -ge 6 ]; then
    echo -e "${GREEN}✅ ${SUBVOL_COUNT}${NC}"
else
    echo -e "${RED}❌ ${SUBVOL_COUNT} (esperado ≥6)${NC}"
    ((FAILED++))
fi

echo -n "BTRFS saudável: "
if btrfs device stats / 2>/dev/null | grep -q "errors 0"; then
    echo -e "${GREEN}✅${NC}"
else
    echo -e "${RED}❌${NC}"
    ((FAILED++))
fi

# Docker
echo ""
echo "═══ 6. DOCKER ═══"
echo -n "Docker service: "
if systemctl is-active docker &> /dev/null; then
    echo -e "${GREEN}✅${NC}"
else
    echo -e "${RED}❌${NC}"
    ((FAILED++))
fi

echo -n "Docker memory limit: "
DOCKER_MEM=$(systemctl show docker.service -p MemoryMax --value 2>/dev/null)
if [[ "$DOCKER_MEM" == "5368709120" ]]; then
    echo -e "${GREEN}✅ 5GB${NC}"
else
    echo -e "${YELLOW}⚠️  $(($DOCKER_MEM/1024/1024/1024))GB${NC}"
fi

# Temperatura
echo ""
echo "═══ 7. TEMPERATURA ═══"
if command -v sensors &> /dev/null; then
    TEMP=$(sensors 2>/dev/null | grep "Core 0" | awk '{print $3}' | tr -d '+°C' || echo "0")
    echo -n "CPU: "
    if (( $(echo "$TEMP > 0 && $TEMP < 85" | bc -l) )); then
        echo -e "${GREEN}${TEMP}°C${NC}"
    else
        echo -e "${RED}${TEMP}°C${NC}"
        ((FAILED++))
    fi
fi

# i3wm
echo ""
echo "═══ 8. INTERFACE ═══"
echo -n "i3wm: "
if command -v i3 &> /dev/null; then
    echo -e "${GREEN}✅${NC}"
else
    echo -e "${RED}❌${NC}"
    ((FAILED++))
fi

# Resumo
echo ""
echo "════════════════════════════════════════════════════════════"
if [ $FAILED -eq 0 ]; then
    echo -e "${GREEN}✅ SISTEMA 100% VALIDADO - PRONTO PRO CLIPFUSION!${NC}"
    exit 0
else
    echo -e "${RED}❌ FALHAS: ${FAILED} - Revise acima${NC}"
    exit 1
fi
Anexo B – Script monitor-system.sh
bash
#!/bin/bash
while true; do
    clear
    echo "╔════════════════════════════════════════════════════════════╗"
    echo "║         DEBIAN TUNADO 3.0 - MONITOR REALTIME             ║"
    echo "╚════════════════════════════════════════════════════════════╝"
    echo ""

    # CPU
    TEMP=$(sensors 2>/dev/null | grep "Core 0" | awk '{print $3}' | tr -d '+°C' || echo "N/A")
    FREQ=$(cat /sys/devices/system/cpu/cpu0/cpufreq/scaling_cur_freq 2>/dev/null)
    if [ -n "$FREQ" ]; then
        FREQ_GHZ=$(echo "scale=2; $FREQ / 1000000" | bc)
    else
        FREQ_GHZ="N/A"
    fi

    echo "┌─ CPU ──────────────────────────────────────────────────────┐"
    echo "│ Freq: ${FREQ_GHZ} GHz | Temp: ${TEMP}°C"
    echo "└────────────────────────────────────────────────────────────┘"
    echo ""

    # Memória
    echo "┌─ MEMÓRIA ──────────────────────────────────────────────────┐"
    free -h | grep -E "Mem|Swap" | awk '{printf "│ %-6s %6s / %6s (usado: %6s)\n", $1, $3, $2, $3}'
    echo "└────────────────────────────────────────────────────────────┘"
    echo ""

    # Swap layers
    echo "┌─ SWAP LAYERS ──────────────────────────────────────────────┐"
    swapon --show --bytes 2>/dev/null | awk 'NR>1 {printf "│ %-20s %8.1f MB (pri: %3s)\n", $1, $3/1024/1024, $5}'
    echo "└────────────────────────────────────────────────────────────┘"
    echo ""

    # Docker
    if systemctl is-active docker &> /dev/null; then
        echo "┌─ DOCKER ───────────────────────────────────────────────────┐"
        CONTAINERS=$(docker ps -q 2>/dev/null | wc -l)
        echo "│ Containers ativos: ${CONTAINERS}"
        if [ "$CONTAINERS" -gt 0 ]; then
            docker stats --no-stream --format "│ {{.Name}}: CPU {{.CPUPerc}} | MEM {{.MemUsage}}" 2>/dev/null
        fi
        echo "└────────────────────────────────────────────────────────────┘"
    fi

    sleep 2
done
Anexo C – Script snapshot-weekly.sh
bash
#!/bin/bash
DATE=$(date +%Y%m%d-%H%M)
KEEP=7

# Snapshots somente leitura
btrfs subvolume snapshot -r / /.snapshots/root-$DATE
btrfs subvolume snapshot -r /home /.snapshots/home-$DATE

echo "$(date): Snapshot $DATE criado" >> /var/log/snapshots.log

# Limpa antigos (mantém últimos 7)
ls -t /.snapshots/root-* 2>/dev/null | tail -n +$(($KEEP+1)) | xargs -r btrfs subvolume delete
ls -t /.snapshots/home-* 2>/dev/null | tail -n +$(($KEEP+1)) | xargs -r btrfs subvolume delete

# Limpeza de logs
journalctl --vacuum-time=7d
